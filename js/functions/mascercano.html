Mas Cercano

const mascercano = (...arguments) => {
    let closest = 0;
    arguments.forEach(item => closest == 0 ? closest = item :
        item > 0 && item <= Math.abs(closest) ? closest = item :
        item < 0 && -item < Math.abs(closest) ? closest = item :
        null
    )
    return closest;
}
